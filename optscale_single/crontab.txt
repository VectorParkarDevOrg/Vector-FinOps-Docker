# Vector FinOps scheduled jobs (replaces ofelia in the multi-container setup)
# All jobs run with HX_ETCD_HOST=127.0.0.1 HX_ETCD_PORT=2379

SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
PYTHONPATH=/usr/src/app

# cleanmongodb: every 3 minutes
*/3 * * * * root cd /usr/src/app && HX_ETCD_HOST=127.0.0.1 HX_ETCD_PORT=2379 python3 docker_images/cleanmongodb/clean-mongo-db.py >> /var/log/optscale/cleanmongodb.log 2>&1
